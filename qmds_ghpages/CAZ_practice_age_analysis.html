Patient data

Loading patient registration data for calculating prescription rates.

Show the code 
patient_age_df <- read_csv("data_raw/practice_patients_age.csv")

patient_avg_age <- as.data.table(patient_age_df)[,
  `:=`(
    lower = as.numeric(str_extract(AGE_GROUP, "(?<=(\\(|\\[))\\d{1,2}")),
    upper = as.numeric(str_extract(AGE_GROUP, "\\d{2,3}(?=(\\)|\\]))"))
  )
][,
  midpoint := 0.5 * (lower + upper)
][,
  .(mean_age = weighted.mean(midpoint, NUMBER_OF_PATIENTS)),
  by = .(ORG_CODE, month, year)
][ORG_CODE %in% pract_classed$code]

patient_avg_age$month <- match(patient_avg_age$month, tolower(month.name))

Variations in mean_age

Show the code 
var_mean_age <- patient_avg_age |>
  mutate(
    ym = paste0(year, sprintf("%02d", month)) |> as.numeric(),
    ym_date = ymd(paste0(ym, "01"))
  ) |>
  mutate(baseline = if_else(ym == min(ym), mean_age, NA), .by = ORG_CODE) |>
  mutate(
    baseline = if_else(is.na(baseline), sum(baseline, na.rm = TRUE), baseline),
    .by = ORG_CODE
  ) |>
  mutate(mean_age_std = mean_age - baseline)


ggplot(var_mean_age, aes(mean_age_std)) +
  stat_ecdf() +
  theme_minimal()

Variation in size

Show the code 
patient_num_df$month <- match(patient_num_df$month, tolower(month.name))

var_size_pract <- patient_num_df |>

  mutate(
    ym = paste0(year, sprintf("%02d", month)) |> as.numeric(),
    ym_date = ymd(paste0(ym, "01"))
  ) |>
  mutate(
    baseline = if_else(ym == min(ym), NUMBER_OF_PATIENTS, NA),
    .by = ORG_CODE
  ) |>
  mutate(
    baseline = if_else(is.na(baseline), sum(baseline, na.rm = TRUE), baseline),
    .by = ORG_CODE
  ) |>
  mutate(patients_std = (NUMBER_OF_PATIENTS - baseline) / baseline)


ggplot(var_size_pract, aes(patients_std)) +
  stat_ecdf() +
  theme_minimal()
